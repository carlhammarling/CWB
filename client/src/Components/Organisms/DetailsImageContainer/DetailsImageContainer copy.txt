import React, { useState } from "react";
import "./DetailsImageContainer.scss";

const DetailsImageContainer = (props: CoworkingSpace) => {
  const [sortedImages, setSortedImages] = useState(props.images);

  console.log(sortedImages);

  const sortImage = (index: number) => {
    const newOrder = [...sortedImages];
    const selectedImage = newOrder[index];
    newOrder.splice(index, 1);
    newOrder.unshift(selectedImage);
    setSortedImages(newOrder);
  };

  return (
    <div className="detailsImageContainer">
      <div
        className="bigImage"
        style={{ backgroundImage: `url(${sortedImages[0]})` }}
      ></div>
      <div className="smallImageContainer">
        {sortedImages.slice(1).map((image, index) => (
          <div
            key={index}
            onClick={() => sortImage(index + 1)}
            className="smallImage"
            style={{ backgroundImage: `url(${image})` }}
          ></div>
        ))}
      </div>
    </div>
  );
};

export default DetailsImageContainer;
